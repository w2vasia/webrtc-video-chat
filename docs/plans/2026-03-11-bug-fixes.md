### Task 8: Restrict rate limiter x-forwarded-for trust

**Files:**
- Modify: `server/src/middleware/rateLimit.ts`
- Modify: `server/src/index.ts`

**Problem:** The rate limiter unconditionally trusts `x-forwarded-for`, which can be forged by any client not behind a verified proxy.

**Step 1: Add `trustProxy` option to rateLimit**

Replace `rateLimit.ts` content:
```ts
import type { Context, Next } from "hono";

interface RateLimitEntry {
  count: number;
  resetAt: number;
}

export function rateLimit(opts: { windowMs: number; max: number; trustProxy?: boolean }) {
  const store = new Map<string, RateLimitEntry>();

  setInterval(() => {
    const now = Date.now();
    for (const [key, entry] of store) {
      if (now >= entry.resetAt) store.delete(key);
    }
  }, 60000);

  return async (c: Context, next: Next) => {
    let ip: string;
    if (opts.trustProxy) {
      ip = c.req.header("x-forwarded-for")?.split(",")[0]?.trim()
        || c.req.header("x-real-ip")
        || "unknown";
    } else {
      ip = c.req.header("x-real-ip") || "unknown";
    }

    const now = Date.now();
    const entry = store.get(ip);

    if (!entry || now >= entry.resetAt) {
      store.set(ip, { count: 1, resetAt: now + opts.windowMs });
      return next();
    }

    entry.count++;
    if (entry.count > opts.max) {
      c.header("Retry-After", String(Math.ceil((entry.resetAt - now) / 1000)));
      return c.json({ error: "Too many requests" }, 429);
    }

    return next();
  };
}
```

Change the rateLimit call to pass `trustProxy`:
